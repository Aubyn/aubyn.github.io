в—Ӣ Once вҖӢ

sync.Once еҸӘдјҡжү§иЎҢдёҖж¬Ў Do и°ғз”Ёзҡ„еҮҪж•°пјҢеҸҜз”ЁжқҘе®һзҺ°еҜ№иұЎеҚ•дҫӢ

go
type User struct {}

var user *User
var once sync.Once

func GetUserSingle() *User {
  once.Do(func() {
    user = &User{}
  })
  return user
}

в—Ӣ Pool вҖӢ

:::warn еҰӮжһңеҜ№иұЎзҡ„еҲӣе»әгҖҒй”ҖжҜҒжҖ§иғҪејҖй”Җе№¶дёҚеӨ§пјҢдё”е…¶з”ҹе‘Ҫе‘Ёжңҹй•ҝпјҢдҪҝз”Ё sync.Pool еҸҜиғҪдјҡеёҰжқҘжӣҙеӨҡиҙҹйқўеҪұе“ҚгҖӮ :::

go
type User struct {}

var UserPool = sync.Pool{
    New: func() any { return &User{} },
}

func GetUserSingle() *User {
    user := UserPool.Get().(*User)
    UserPool.Put(user)
    return user
}
